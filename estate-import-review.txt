   123	        if options["limit"] is not None:

   175	                            listing_key=listing_key

   180	                                listing_key=listing_key,
   181	                                **defaults,
   182	                            )
   183	                            counters["created"] += 1
   184	                            action = "CREATE"
   185	                        else:
   186	                            changed_fields = [
   187	                                field_name
   188	                                for field_name, value in defaults.items()
   189	                                if getattr(existing, field_name) != value
   190	                            ]
   191	
   192	                            if not changed_fields:
   193	                                counters["unchanged"] += 1
   194	                                action = "UNCHANGED"
   195	                            else:
   196	                                for field_name in changed_fields:
   197	                                    setattr(existing, field_name, defaults[field_name])
   198	                                existing.save(update_fields=changed_fields)
   199	                                counters["updated"] += 1
   200	                                action = "UPDATE"
   201	
   202	                        if options["verbosity"] >= 2:
   203	                            self.stdout.write(
   204	                                f"{action:9} {listing_key}  {title}"

   215	                transaction.set_rollback(True)

   217	        counters["mapping_failed"] = mapping_failures
   218	        self._print_database_summary(counters, dry_run=dry_run)

   226	            raise CommandError(f"Could not read XML: {exc}") from exc

   514	    def _collect_meta(self, item: ET.Element) -> dict[str, Any]:
   515	        meta: dict[str, Any] = {}
   516	
   517	        for postmeta in item.findall("wp:postmeta", NS):
   518	            key = self._text(postmeta, "wp:meta_key")
   519	            if not key:
   520	                continue
   521	            value = self._text(postmeta, "wp:meta_value")
   522	
   523	            if key not in meta:
   524	                meta[key] = value
   525	            elif isinstance(meta[key], list):
   526	                meta[key].append(value)
   527	            else:
   528	                meta[key] = [meta[key], value]
   529	
   530	        return meta

   532	    def _collect_terms(
   533	        self, item: ET.Element
   534	    ) -> tuple[dict[str, list[str]], dict[str, list[dict[str, str]]]]:
   535	        canonical: dict[str, list[str]] = {
   536	            "type": [],
   537	            "status": [],
   538	            "features": [],

   102	        previous = snapshot.imported_project_values if snapshot else {}
   103	        imported_values = {}
   104	        changed_fields = []
   105	
   106	        for field_name, incoming_value in incoming.items():
   107	            current_value = getattr(project, field_name)
   108	            current_json = _json_value(current_value)

   297	def parse_estate_content(raw_html: str, metadata=None, terms=None) -> ParsedEstateContent:
   298	    result = ParsedEstateContent()
   299	    units = _supporting_inputs(result, metadata, terms)
   300	    collector = _RawCollector()
   301	    collector.feed(raw_html or "")
   302	    collector.close()
   303	
   304	    heading = "Overview"
   305	    kind = None
   306	    heading_key = "overview"
   307	    heading_occurrences: dict[str, int] = {}
   308	    residual: list[str] = []
   309	    deposits: list[ParsedItem] = []
   310	    section_unit_key = ""
   311	
   312	    def flush():
   313	        nonlocal residual, deposits
   314	        if residual and "".join(residual).strip():
   315	            occurrence = heading_occurrences.get(heading_key, 0)
   316	            result.sections.append(ParsedSection(f"html-section:{heading_key}:{occurrence}", heading, "".join(residual)))
   317	        if deposits:
   318	            occurrence = heading_occurrences.get(heading_key, 0)
   319	            result.deposit_plans.append(ParsedDepositPlan(f"html-deposit-plan:{heading_key}:{occurrence}", heading or "Deposit plan", tuple(deposits), section_unit_key))
   320	        residual, deposits = [], []
   321	
   322	    seen_documents = {item.source_key for item in result.documents}
   323	    counters: dict[str, int] = {}
   324	
   325	    for root_tag, original_fragment in collector.fragments:
   326	        fragment = original_fragment
   327	        fragment_text = _text(fragment)
   328	        if root_tag in {"h1", "h2", "h3", "h4", "h5", "h6"}:
   329	            flush()
   330	            heading = fragment_text or "Overview"
   331	            kind = _kind(heading)
   332	            heading_key = slugify(heading)[:80] or "section"
   333	            heading_occurrences[heading_key] = heading_occurrences.get(heading_key, -1) + 1
   334	            section_unit_key = _matching_unit(heading, units, result.warnings)
   335	            continue
   336	
   337	        # Extract protected document anchors and remove only those exact anchors
   338	        # from public residual HTML. Other links and all surrounding markup remain.
   339	        for anchor in _elements(fragment, {"a"}):
   340	            anchor_links = _links(anchor)
   341	            if not anchor_links:
   342	                continue
   343	            label, url = anchor_links[0]
   344	            if _valid_url(url) and DOCUMENT_PATTERN.search(f"{label} {url}"):
   345	                source_key = _key("html-document", url)
   346	                if source_key not in seen_documents:
   347	                    result.documents.append(ParsedItem(source_key, label or "Document", url=url))
   348	                    seen_documents.add(source_key)
   349	                fragment = fragment.replace(anchor, "", 1)
   350	
   351	        candidate_fragments = _elements(fragment, {"tr"}) or _elements(fragment, {"li"})
   352	        candidates = candidate_fragments or ([fragment] if fragment_text.strip() else [])
   353	        if not kind:
   354	            residual.append(fragment)
   355	            continue
   356	        if kind == "documents":
   357	            if _text(fragment).strip():
   358	                residual.append(fragment)
   359	                result.warnings.append(f"Unsupported document-section content preserved under '{heading}'.")
   360	            continue
   361	
   362	        parsed_items: list[ParsedItem] = []
   363	        ambiguous = False
   364	        for candidate in candidates:
   365	            text = _text(candidate).strip(" |")
   366	            if not text:
   367	                continue
   368	            if _links(candidate):
   369	                ambiguous = True
   370	                break
   371	            parts = [part.strip() for part in re.split(r"\s*//+\s*", text) if part.strip()]
   372	            if kind == "prices":
   373	                display = " // ".join(parts)
   374	                if not re.search(r"\$|\bCAD\b|\bfrom\b|\bstarting\b", display, re.I):
   375	                    ambiguous = True
   376	                    break
   377	                amount = normalize_money(display)
   378	                if amount is None:
   379	                    result.warnings.append(f"Ambiguous price retained as display text: {display[:160]}")
   380	                association = _matching_unit(display, units, result.warnings)
   381	                counter_key = f"price:{heading_key}"
   382	                counters[counter_key] = counters.get(counter_key, 0) + 1
   383	                parsed_items.append(ParsedItem(_key(f"html-price-{heading_key}-{counters[counter_key]}", association or parts[0]), display, amount, unit_type_key=association))
   384	            elif kind == "unit_types":
   385	                for part in parts:
   386	                    if re.search(r"\$|%", part) or len(part) > 180:
   387	                        ambiguous = True
   388	                        break
   389	                    source_key = _key("html-unit", part)
   390	                    item = ParsedItem(source_key, part)
   391	                    if source_key not in units:
   392	                        parsed_items.append(item)
   393	                        units[source_key] = item
   394	            elif kind == "deposits":
   395	                amount, percentage = normalize_money(text), normalize_percentage(text)
   396	                if amount is None and percentage is None:
   397	                    ambiguous = True
   398	                    break
   399	                milestone = text.split(":", 1)[0].strip() if ":" in text else text
   400	                counter_key = f"deposit:{heading_key}:{milestone}"
   401	                counters[counter_key] = counters.get(counter_key, 0) + 1
   402	                parsed_items.append(ParsedItem(_key(f"html-deposit-{heading_key}-{counters[counter_key]}", milestone), text, amount, percentage, milestone, unit_type_key=section_unit_key))
   403	            elif kind in {"incentives", "amenities"}:
   404	                if len(text) > 500:
   405	                    ambiguous = True
   406	                    break
   407	                parsed_items.append(ParsedItem(_key(f"html-{kind[:-1]}", text), text))
   408	
   409	        if ambiguous:
   410	            residual.append(fragment)
   411	            result.warnings.append(f"Ambiguous or unsupported {kind} content preserved under '{heading}'.")
   412	            continue
   413	        if kind == "deposits":
   414	            deposits.extend(parsed_items)
   415	        else:
   416	            getattr(result, kind).extend(parsed_items)
   417	
   418	    flush()
   419	    if raw_html and not collector.fragments:
   420	        result.sections.append(ParsedSection("html-section:overview:0", "Overview", raw_html))
   421	        result.warnings.append("Source HTML could not be segmented and was preserved verbatim.")
   422	    for field_name in ("unit_types", "prices", "incentives", "amenities", "documents"):
   423	        values = getattr(result, field_name)
   424	        setattr(result, field_name, list({item.source_key: item for item in values}.values()))
   425	    return result

     5	def normalize_money(text: str) -> Decimal | None:
     6	    """Normalize only an explicit, single dollar amount."""
     7	    matches = re.findall(r"(?:CAD\s*)?\$\s*([\d,]+(?:\.\d{1,2})?)", text, re.I)
     8	    if len(matches) != 1:
     9	        return None
    10	    try:
    11	        return Decimal(matches[0].replace(",", ""))
    12	    except InvalidOperation:
    13	        return None

    16	def normalize_percentage(text: str) -> Decimal | None:
    17	    matches = re.findall(r"(?<!\d)(\d{1,3}(?:\.\d+)?)\s*%", text)
    18	    if len(matches) != 1:
    19	        return None
    20	    value = Decimal(matches[0])
    21	    return value if value <= 100 else None
