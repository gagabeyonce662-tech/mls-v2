    77	function richTextToPlainText(value: unknown): string {
    78	  const html = String(value ?? "").trim();
    79	  if (!html) return "";
    80	
    81	  return html
    82	    .replace(/<script\b[^>]*>[\s\S]*?<\/script>/gi, " ")
    83	    .replace(/<style\b[^>]*>[\s\S]*?<\/style>/gi, " ")
    84	    .replace(/<br\s*\/?>/gi, "\n")
    85	    .replace(/<\/p\s*>/gi, "\n\n")
    86	    .replace(/<\/h[1-6]\s*>/gi, "\n\n")
    87	    .replace(/<li\b[^>]*>/gi, "• ")
    88	    .replace(/<\/li\s*>/gi, "\n")
    89	    .replace(/<[^>]+>/g, " ")
    90	    .replace(/&nbsp;/gi, " ")
    91	    .replace(/&amp;/gi, "&")
    92	    .replace(/&quot;/gi, '\"')
    93	    .replace(/&#39;|&apos;/gi, "'")
    94	    .replace(/[ \t]+\n/g, "\n")
    95	    .replace(/\n[ \t]+/g, "\n")
    96	    .replace(/[ \t]{2,}/g, " ")
    97	    .replace(/\n{3,}/g, "\n\n")
    98	    .trim();
    99	}

   102	 * Dynamic Metadata Generation
   103	 * Optimizes SEO and Social Sharing (OpenGraph/Twitter) for each property listing.

   105	export async function generateMetadata({
   106	  params,
   107	}: ListingPageProps): Promise<Metadata> {
   108	  const id = (await params).id;
   109	  const property = await getEstateProperty(id);
   110	
   111	  if (!property) {
   112	    return {
   113	      title: "Listing not found | Estate-4u",
   114	      robots: { index: false, follow: false },
   115	    };
   116	  }
   117	
   118	  const metaPriv = getListingIsPrivileged();
   119	  const address =
   120	    String(property.unparsed_address || property.address || "").trim() ||
   121	    getDisplayAddress(property, { isPrivileged: metaPriv }) ||
   122	    "Property Details";
   123	  const rawCity = String(property.city || property.City || "").trim();
   124	  const city = ["unknown city", "n/a"].includes(rawCity.toLowerCase())
   125	    ? ""
   126	    : rawCity;
   127	  const description =
   128	    getDescription(property) ||
   129	    `View photos and property details for ${address} on Estate-4u.`;
   130	  const images = getPhotos(property).filter(Boolean).slice(0, 5);
   131	
   132	  return {
   133	    title: `${address} | ${city ? `${city} Real Estate` : "Real Estate"}`,
   134	    description: description.substring(0, 160),
   135	    openGraph: {
   136	      title: `${address} | Estate-4u`,
   137	      description: description.substring(0, 160),
   138	      ...(images.length > 0 ? { images } : {}),
   139	    },
   140	    twitter: {
   141	      card: "summary_large_image",
   142	      title: `${address} | Estate-4u`,
   143	      description: description.substring(0, 160),
   144	      ...(images[0] ? { images: [images[0]] } : {}),
   145	    },
   146	  };
   147	}

    13	  listingKey: string;
    14	  title?: string;
    15	}) {
    16	  const { user, isLoading } = useUserAuth();
    17	  const [body, setBody] = useState("");
    18	  const [savedAt, setSavedAt] = useState<string | null>(null);
    19	  const [loading, setLoading] = useState(false);
    20	  const [saving, setSaving] = useState(false);
    21	  const [error, setError] = useState<string | null>(null);
    22	
    23	  const authHeader = useCallback((): Record<string, string> => {
    24	    const token =
    25	      typeof window !== "undefined"
    26	        ? window.localStorage.getItem("access_token")
    27	        : null;
    28	    return token ? { Authorization: `Bearer ${token}` } : {};
    29	  }, []);
    30	
    31	  useEffect(() => {
    32	    if (!user || !listingKey) return;
    33	    setLoading(true);
    34	    setError(null);
    35	    const base = API_BASE_URL.replace(/\/+$/, "");
    36	    fetch(
    37	      `${base}/api/mls/property-notes/?listing_key=${encodeURIComponent(listingKey)}`,
    38	      { headers: { ...authHeader() } },
    39	    )
    40	      .then((r) => {
    41	        if (!r.ok) throw new Error("load");
    42	        return r.json();
    43	      })
    44	      .then((d) => {
    45	        setBody(String(d.body ?? ""));
    46	        setSavedAt(d.updated_at ? String(d.updated_at) : null);
    47	      })
    48	      .catch(() => setError("Could not load notes."))
    49	      .finally(() => setLoading(false));
    50	  }, [user, listingKey, authHeader]);
    51	
    52	  const save = () => {
    53	    if (!user) return;
    54	    setSaving(true);
    55	    setError(null);
    56	    const base = API_BASE_URL.replace(/\/+$/, "");
    57	    fetch(`${base}/api/mls/property-notes/`, {
    58	      method: "PUT",
    59	      headers: {
    60	        "Content-Type": "application/json",
    61	        ...authHeader(),
    62	      },
    63	      body: JSON.stringify({ listing_key: listingKey, body }),
    64	    })
    65	      .then((r) => {
    66	        if (!r.ok) throw new Error("save");
    67	        return r.json();
    68	      })
    69	      .then((d) => {
    70	        setSavedAt(d.updated_at ? String(d.updated_at) : null);
    71	      })
    72	      .catch(() => setError("Could not save notes."))
    73	      .finally(() => setSaving(false));
    74	  };
    75	
    76	  if (isLoading) return null;
    77	
    78	  if (!user) {
    79	    return (
    80	      <section className="bg-white border border-ds-card-border rounded-2xl p-6 shadow-sm">
    81	        <h2 className={`${ds.h3} mb-2`}>{title}</h2>

    87	          className="text-sm font-medium text-ds-primary hover:underline"
    88	        >
    89	          Sign in
    90	        </Link>
    91	      </section>
    92	    );
    93	  }

    96	    <section className="bg-white border border-ds-card-border rounded-2xl p-6 shadow-sm">

    99	        Visible only to your account. Not shared with listing agents.
   100	      </p>

   102	      <textarea
